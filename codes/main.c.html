#include <stdio.h>


/** 
 * 质数又称素数。一个大于 1 的自然数，除了 1 和它自身外，不能被其他 
 * 自然数整除的数叫做质数；否则称为合数（规定 1 既不是质数也不是合数）。 
 */
int main_prime() {
    int i, num;
    scanf("%d", &num);
    for (i=2; i<=num/2; i++) {
        if (i % num == 0)
            break;
    }
    if (i>num) {
        printf("%d is prime", num);
    } else {
        printf("%d is not prime", num);
    }
    return 0;
}

/** 
 * 一个饲养场引进一只刚出生的新品种兔子，这种兔子从出生的下一个月开始，每月新生 
 * 一只兔子，新生的兔子也如此繁殖.如果所有的兔子都不死去，问到第12个月时，该饲 
 * 养场共有兔子多少只？ 
 */
int main1() {
    int i, count0, count1;
    for (i=2, count0=1; i<=12; i++) {
        count1 = count0 * 2;
        count0 = count1;
    }
    printf("%d\n", count1);
    return 0;
}

/** 
 * 验证谷角猜想。日本数学家谷角静夫在研究自然数时发现了一个奇怪现象:对于任意一 
 * 个自然数n，若n为偶数，则将其除以2﹔若n为奇数，则将其乘以3，然后再加1。如 
 * 此经过有限次运算后，总可以得到自然数1。人们把谷角静夫的这一发现叫做“谷角猜想”。 
 */
int main2() {
    int num;
    scanf("%d", &num);
    while (num != 1) {
        if (num % 2 == 0) {
            printf("%d, %d\n", num, num / 2);
            num = num / 2;
        } else {
            printf("%d, %d\n", num, num * 3 + 1);
            num = num * 3 + 1;
        }
    }
    return 0;
}

/** 
 * 利用辗转相除法求解两个数的最大公约数和最小公倍数。 
 */
int main3() {
    int m, m1, n, n1, t, r;
    scanf("%d,%d", &m, &n);
    m1 = m;
    n1 = n;
    if (m < n) {
        t = m;
        m = n;
        n = t;
    }
    while (r = m % n) {
        m = n;
        n = r;
    }
    printf("%d\n", n);
    printf("%d\n", m1 * n1 / n);
    return 0;
}

/** 
 * 用迭代法求x = √a。 
 * 求平方根的迭代公式为x[n+1]=1/2(x[n]+a/x[n]) 
 * 要求前后两次求出的x的差的绝对值小于10^-5。 
 */
#define F(x, a) ((x + a / x) / 2)
#include <math.h>
int main4() {
    float a, x0, x1;
    scanf("%f", &a);
    x0 = a / 2;
    x1 = F(x0, a);
    do {
        x0 = x1;
        x1 = F(x0, a);
    } while (fabs((double)(x0 - x1)) >= 1e-5);
    printf("%g\n", x1);
    return 0;
}
#undef F
/* ************************************************************************** */
#define F(x,a) (x*x-a)
#define F1(x) (2*x)

        x1 = x0;

    } while (fabs((double)(x0 - x1)) >= 1e-6);

}

int main4_2() {
    float a, left, right, mid;
    scanf("%f", &a);
    left = 0;
    right = a;
    do {
        mid = (left + right) / 2;
        if (mid * mid > a) {
            right = mid;
        } else {
            left = mid;
        }
    } while (fabs((double)(left - right)) >= 1e-6);
    printf("%g\n", (left + right) / 2);
    return 0;
}

/** 
 * 用牛顿迭代法求下面方程在1.5附近的根: 
 * 2x^3-4x^2+3x-6=0 
 * 牛顿迭代法的迭代公式是:x[n+1]=x[n]-f(x[n])/f'(x[n]). 
 */
#define F(x) (2*x*x*x-4*x*x+3*x-6)
#define F1(x) (6*x*x-8*x+3)
int main5() {
    float x0, x1;
    x0 = 1.5f;
    do {
        x1 = x0;
        x0 = x1 - F(x1) / F1(x1);
    } while (fabs((double)(x0 - x1)) >= 1e-6);
    printf("%g\n", x0);
    return 0;
}
#undef F
#undef F1

/** 
 * 用二分法求下面方程在(-10,10)之间的根： 
 * 2x^3-4x^2+3x-6=0 
 */
#define F(x) (2*x*x*x-4*x*x+3*x-6)
int main6() {
    float left, right, mid;
    left = -10;
    right = 10;
    if (F(left) * F(right)) {
        do {
            mid = (left + right) / 2;
            if (F(left) * F(mid) < 0) {
                right = mid;
            } else {
                left = mid;
            }
        } while (fabs((double)(left - right)) >= 1e-6);
        printf("%g, %g\n", left, right);
    }
    return 0;
}
#undef F
