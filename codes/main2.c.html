218  int main_bubble_sort() {
219      int a[]={91,67,35,62,29,72,46};
220      int i,j,temp;
221      for (i=0; i<7; i++) {
222          for (j=i+1; j<7; j++) {
223              if (a[i] > a[j]) {
224                  temp = a[j];
225                  a[j] = a[i];
226                  a[i] = temp;
227              }
228          }
229      }
230      for (i=0; i<7; i++)
231          printf("%3d", a[i]);
232      return 0;
233  }

270          for (j=0;j<4;j++)
271              printf("%3d", c[i][j]);

280  int main_matrix_mul() {
281      int a[3][2]={2,-1,
282                   -4,0,
283                   3,1},
284          b[2][2]={7,-9,
285                   -8,10};
286      int i,j,k,s,c[3][2];
287      for (i=0; i<3; i++)
288          for (j=0; j<2; j++) {
289              for (k=s=0; k<2; k++)
290                  s+=a[i][k]*b[k][j];
291              c[i][j]=s;
292          }
293      for (i=0;i<3;i++) {
294          for (j=0;j<2;j++)
295              printf("%6d", c[i][j]);
296          printf("\n");
297      }
298      return 0;
299  }

304  int main_matrix_symmetric_main() {
305      int a[4][4]={1,2,3,4,
306                   2,2,5,6,
307                   3,5,3,7,
308                   4,6,7,4};
309      int i,j,found=0;
310      for (j=0;j<4;j++)
311          for (i=j;i<4;i++)
312              if (a[j][i]!=a[i][j]) {
313                  found = 1;
314                  break;
315              }
316      if (found==1)
317          printf("No");
318      else
319          printf("Yes");
320      return 0;
321  }
